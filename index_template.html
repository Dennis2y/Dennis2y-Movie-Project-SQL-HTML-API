from jinja2 import Template def generate_website(movies): """ Generates an HTML page displaying a list of movies. The movies should be a dictionary with the structure: { 'title': {'year': int, 'rating': float, 'poster_url': str} } """ # Define the template (could also load from a file) html_template = """ Movies List

Movies List

{% for title, movie in movies.items() %}

{{ title }}

Year: {{ movie.year }}

Rating: {{ movie.rating }}

{% if movie.poster_url %} {% endif %}
{% endfor %}
""" # Create a Jinja2 Template object template = Template(html_template) # Render the template with the provided movie data html_output = template.render(movies=movies) # Write the HTML to a file (e.g., movies_list.html) with open('movies_list.html', 'w') as f: f.write(html_output) print("Website generated: movies_list.html")
